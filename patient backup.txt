session_start();

// Check if user is logged in
if (!isset($_SESSION['user_id'])) {
    header("Location: ../login.php");
    exit;
}

// Include database connection
require_once('../includes/db_connection.php');

// Fetch user profile information for the logged-in user
$user_id = $_SESSION['user_id'];
$sql = "SELECT * FROM users WHERE id = $user_id";
$result = mysqli_query($conn, $sql);

if ($result && mysqli_num_rows($result) > 0) {
    // Retrieve user profile information
    $user_info = mysqli_fetch_assoc($result);

    // Check if the user is a patient
    if ($user_info['user_type'] === 'patient') {
        // Fetch appointments for the logged-in patient
        $sql = "SELECT * FROM appointments WHERE patient_id = $user_id";
        $result = mysqli_query($conn, $sql);

        if ($result && mysqli_num_rows($result) > 0) {
            // Retrieve appointments
            $appointments = mysqli_fetch_all($result, MYSQLI_ASSOC);
        } else {
            // Handle error if no appointments found
            $error_message = "No appointments scheduled.";
        }

        // Fetch lab reports for the logged-in patient
        $sql = "SELECT * FROM lab_reports WHERE patient_id = $user_id";
        $result = mysqli_query($conn, $sql);

        if ($result && mysqli_num_rows($result) > 0) {
            // Retrieve lab reports
            $lab_reports = mysqli_fetch_all($result, MYSQLI_ASSOC);
        } else {
            // Handle error if no lab reports found
            $error_message = "No lab reports available.";
        }

        // Fetch billing information for the logged-in patient
        $sql = "SELECT * FROM billing WHERE patient_id = $user_id";
        $result = mysqli_query($conn, $sql);

        if ($result && mysqli_num_rows($result) > 0) {
            // Retrieve billing information
            $billing_info = mysqli_fetch_assoc($result);
        } else {
            // Handle error if no billing information found
            $error_message = "No billing information available.";
        }
    } else {
        // Redirect to appropriate dashboard based on user type
        if ($user_info['user_type'] === 'doctor') {
            header("Location: doctor_dashboard.php");
            exit;
        } elseif ($user_info['user_type'] === 'technician') {
            header("Location: technician_dashboard.php");
            exit;
        }
    }
} else {
    // Handle error if no user found
    $error_message = "User not found.";
}

// Close database connection
mysqli_close($conn);
